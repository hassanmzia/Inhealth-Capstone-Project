from django.http import JsonResponse
from django.views.decorators.csrf import csrf_exempt
import json

class MCPServer:
    """
    Model Context Protocol Server
    Provides standardized context and tools to LLM agents
    """
    
    def __init__(self):
        self.tool_registry = ToolRegistry()
        self.context_manager = ContextManager()
    
    @csrf_exempt
    def handle_context_request(self, request):
        """
        Endpoint: POST /mcp/context
        Returns MCP-formatted context for agent
        """
        data = json.loads(request.body)
        patient_id = data.get('patient_id')
        agent_id = data.get('agent_id')
        
        # Build MCP context
        context = {
            "protocol": "MCP/1.0",
            "version": "2024-10",
            "timestamp": datetime.now().isoformat(),
            "context": {
                # Patient clinical context
                "patient": self._get_patient_context(patient_id),
                
                # Conversation history (for multi-turn reasoning)
                "conversation_history": self._get_conversation_history(
                    patient_id, agent_id
                ),
                
                # Available tools this agent can use
                "available_tools": self.tool_registry.get_tools_for_agent(agent_id),
                
                # Clinical constraints and safety rules
                "constraints": self._get_clinical_constraints(patient_id),
                
                # Retrieved knowledge (RAG results if any)
                "knowledge_base": self._get_relevant_knowledge(patient_id, agent_id),
                
                # Inter-agent context (what other agents have found)
                "agent_findings": self._get_agent_findings(patient_id)
            },
            
            # Metadata
            "metadata": {
                "requesting_agent": agent_id,
                "patient_consent": self._verify_consent(patient_id),
                "data_sources": ["FHIR", "Neo4j", "Qdrant"],
                "compliance": ["HIPAA", "21 CFR Part 11"]
            }
        }
        
        return JsonResponse(context)
    
    def _get_patient_context(self, patient_id):
        """Retrieve comprehensive patient context from all databases"""
        from apps.fhir.models import FHIRPatient, FHIRObservation, FHIRCondition
        
        patient = FHIRPatient.objects.get(id=patient_id)
        
        return {
            "demographics": {
                "id": str(patient.id),
                "mrn": patient.identifier,
                "name": f"{patient.name_given} {patient.name_family}",
                "age": patient.calculate_age(),
                "gender": patient.gender
            },
            
            # Active conditions
            "conditions": [
                {
                    "code": c.code_code,
                    "display": c.code_display,
                    "status": c.clinical_status,
                    "onset": c.onset_datetime.isoformat() if c.onset_datetime else None
                }
                for c in FHIRCondition.objects.filter(
                    patient=patient, 
                    clinical_status='active'
                )
            ],
            
            # Current medications
            "medications": self._get_medications(patient),
            
            # Recent vital signs (last 7 days)
            "recent_vitals": self._get_recent_vitals(patient),
            
            # Lab results
            "lab_results": self._get_lab_results(patient),
            
            # Family history (from Neo4j)
            "family_history": self._get_family_history(patient_id),
            
            # Risk scores (from ML models)
            "risk_scores": self._get_risk_scores(patient_id)
        }
    
    def _get_conversation_history(self, patient_id, agent_id):
        """
        Retrieve multi-turn conversation history
        Important for maintaining context across agent interactions
        """
        from apps.agents.models import ConversationHistory
        
        history = ConversationHistory.objects.filter(
            patient_id=patient_id,
            agent_id=agent_id
        ).order_by('-timestamp')[:10]  # Last 10 interactions
        
        return [
            {
                "role": h.role,  # "agent" or "system"
                "content": h.content,
                "timestamp": h.timestamp.isoformat(),
                "actions_taken": h.actions_taken
            }
            for h in history
        ]
    
    @csrf_exempt
    def handle_tool_execution(self, request):
        """
        Endpoint: POST /mcp/tools/execute
        Execute tools on behalf of LLM
        """
        data = json.loads(request.body)
        tool_name = data.get('tool_name')
        parameters = data.get('parameters')
        agent_id = data.get('agent_id')
        
        # Verify agent has permission to use this tool
        if not self.tool_registry.can_agent_use_tool(agent_id, tool_name):
            return JsonResponse({
                "error": "Permission denied",
                "message": f"Agent {agent_id} cannot use tool {tool_name}"
            }, status=403)
        
        # Execute tool
        try:
            tool = self.tool_registry.get_tool(tool_name)
            result = tool.execute(**parameters)
            
            return JsonResponse({
                "protocol": "MCP/1.0",
                "tool_name": tool_name,
                "result": result,
                "execution_time": tool.execution_time,
                "success": True
            })
            
        except Exception as e:
            return JsonResponse({
                "protocol": "MCP/1.0",
                "tool_name": tool_name,
                "error": str(e),
                "success": False
            }, status=500)

class MCPTool:
    """Represents a single MCP-compliant tool"""
    
    def __init__(self, name, description, parameters, executor, allowed_agents):
        self.name = name
        self.description = description
        self.parameters = parameters
        self.executor = executor
        self.allowed_agents = allowed_agents
        self.execution_time = None
    
    def execute(self, **kwargs):
        """Execute the tool with given parameters"""
        import time
        start_time = time.time()
        
        # Validate parameters
        self._validate_parameters(kwargs)
        
        # Execute
        result = self.executor(**kwargs)
        
        self.execution_time = f"{(time.time() - start_time) * 1000:.2f}ms"
        
        return result
    
    def to_mcp_format(self):
        """Return tool definition in MCP format"""
        return {
            "type": "function",
            "function": {
                "name": self.name,
                "description": self.description,
                "parameters": self.parameters
            }
        }
