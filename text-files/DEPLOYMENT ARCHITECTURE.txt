# docker-compose.yml

version: '3.8'

services:
  # PostgreSQL with FHIR schema
  postgres:
    image: postgres:15
    environment:
      POSTGRES_DB: healthcare_fhir
      POSTGRES_USER: healthcare
      POSTGRES_PASSWORD: secure_password
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./init_fhir_schema.sql:/docker-entrypoint-initdb.d/init.sql
  
  # Neo4j Graph Database
  neo4j:
    image: neo4j:5.12
    environment:
      NEO4J_AUTH: neo4j/password
      NEO4J_PLUGINS: '["apoc", "graph-data-science"]'
    ports:
      - "7474:7474"  # Web UI
      - "7687:7687"  # Bolt protocol
    volumes:
      - neo4j_data:/data
  
  # Qdrant Vector Database
  qdrant:
    image: qdrant/qdrant:latest
    ports:
      - "6333:6333"
    volumes:
      - qdrant_data:/qdrant/storage
  
  # Redis (Celery broker)
  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
  
  # Django Application
  django:
    build: .
    command: gunicorn config.wsgi:application --bind 0.0.0.0:8000 --workers 4
    volumes:
      - .:/app
      - static_volume:/app/staticfiles
      - media_volume:/app/media
    ports:
      - "8000:8000"
    environment:
      - DATABASE_URL=postgresql://healthcare:secure_password@postgres:5432/healthcare_fhir
      - NEO4J_URI=bolt://neo4j:7687
      - QDRANT_HOST=qdrant
      - REDIS_URL=redis://redis:6379/0
      - LLAMA_MODEL_PATH=/app/ml_models/llama-3.2-3b
    depends_on:
      - postgres
      - neo4j
      - qdrant
      - redis
  
  # Celery Worker (for agents)
  celery_worker:
    build: .
    command: celery -A config worker -l info --concurrency=10
    volumes:
      - .:/app
    environment:
      - DATABASE_URL=postgresql://healthcare:secure_password@postgres:5432/healthcare_fhir
      - NEO4J_URI=bolt://neo4j:7687
      - QDRANT_HOST=qdrant
      - REDIS_URL=redis://redis:6379/0
    depends_on:
      - redis
      - postgres
  
  # Celery Beat (scheduler)
  celery_beat:
    build: .
    command: celery -A config beat -l info --scheduler django_celery_beat.schedulers:DatabaseScheduler
    volumes:
      - .:/app
    environment:
      - REDIS_URL=redis://redis:6379/0
      - DATABASE_URL=postgresql://healthcare:secure_password@postgres:5432/healthcare_fhir
    depends_on:
      - redis
      - postgres
  
  # Llama 3.2 LLM Server (Ollama)
  llama:
    image: ollama/ollama:latest
    ports:
      - "11434:11434"
    volumes:
      - ollama_data:/root/.ollama
    command: serve
  
  # Nginx (reverse proxy)
  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf
      - static_volume:/app/staticfiles
      - media_volume:/app/media
    depends_on:
      - django

volumes:
  postgres_data:
  neo4j_data:
  qdrant_data:
  ollama_data:
  static_volume:
  media_volume:
