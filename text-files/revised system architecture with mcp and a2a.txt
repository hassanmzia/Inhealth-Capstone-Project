│  │  │    # MCP Protocol Implementation                                 │ │  │
│  │  │    def get_mcp_context(self, patient_id):                        │ │  │
│  │  │      """Retrieve MCP-formatted context"""                        │ │  │
│  │  │      return {                                                    │ │  │
│  │  │        "protocol": "MCP/1.0",                                   │ │  │

│  │  │        }                                                         │ │  │
│  │  │      }                                                           │ │  │
│  │  │                                                                   │ │  │
│  │  │    # A2A Protocol Implementation                                 │ │  │
│  │  │    def send_a2a_message(self, recipient_id, message_type,       │ │  │
│  │  │                         payload, priority="NORMAL"):             │ │  │
│  │  │      """Send A2A protocol message to another agent"""           │ │  │
│  │  │      message = {                                                │ │  │
│  │  │        "protocol": "A2A/1.0",                                   │ │  │
│  │  │        "sender": {"agent_id": self.agent_id},                  │ │  │
│  │  │        "recipient": {"agent_id": recipient_id},                │ │  │
│  │  │        "message_type": message_type,                           │ │  │
│  │  │        "priority": priority,                                    │ │  │
│  │  │        "payload": payload                                       │ │  │
│  │  │      }                                                           │ │  │
│  │  │      self.message_bus.publish(                                  │ │  │
│  │  │        f"agent.{recipient_id}", message                        │ │  │
│  │  │      )                                                           │ │  │
│  │  │                                                                   │ │  │
│  │  │    def receive_a2a_message(self, message):                       │ │  │
│  │  │      """Handle incoming A2A message"""                           │ │  │
│  │  │      if message["message_type"] == "ALERT":                     │ │  │
│  │  │        self.handle_alert(message["payload"])                    │ │  │
│  │  │      elif message["message_type"] == "REQUEST":                 │ │  │
│  │  │        self.handle_request(message["payload"])                  │ │  │
│  │  │      elif message["message_type"] == "RESPONSE":                │ │  │
│  │  │        self.handle_response(message["payload"])                 │ │  │
