def enhance_image(image, defects):
      """
      Apply corrections for fixable issues
      """
      enhanced = image.copy()
      
      # Contrast enhancement for under/overexposure
      if defects['underexposed'] > 0.5 or defects['overexposed'] > 0.5:
          enhanced = apply_clahe(enhanced)  # Adaptive histogram equalization
      
      # Noise reduction
      if defects['noise'] > 0.5:
          enhanced = apply_bilateral_filter(enhanced)
      
      # Sharpening (mild, to avoid amplifying noise)
      enhanced = apply_unsharp_mask(enhanced, amount=0.3)
      
      return enhanced
