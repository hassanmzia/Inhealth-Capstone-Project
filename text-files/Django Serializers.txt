# apps/patients/serializers.py

from rest_framework import serializers
from .models import FHIRPatient, FHIRObservation, FHIRCondition, FHIRMedicationRequest

class PatientSerializer(serializers.ModelSerializer):
    age = serializers.SerializerMethodField()
    active_conditions = serializers.SerializerMethodField()
    current_medications = serializers.SerializerMethodField()
    
    class Meta:
        model = FHIRPatient
        fields = ['id', 'identifier', 'name_family', 'name_given', 'birth_date', 
                  'age', 'gender', 'telecom_phone', 'telecom_email', 
                  'active_conditions', 'current_medications']
    
    def get_age(self, obj):
        from datetime import date
        today = date.today()
        return today.year - obj.birth_date.year - ((today.month, today.day) < (obj.birth_date.month, obj.birth_date.day))
    
    def get_active_conditions(self, obj):
        conditions = FHIRCondition.objects.filter(patient=obj, clinical_status='active')
        return [c.code_display for c in conditions]
    
    def get_current_medications(self, obj):
        meds = FHIRMedicationRequest.objects.filter(patient=obj, status='active')
        return [m.medication_display for m in meds]

class ObservationSerializer(serializers.ModelSerializer):
    class Meta:
        model = FHIRObservation
        fields = ['id', 'code_display', 'value_quantity', 'value_unit', 
                  'effective_datetime', 'interpretation']
